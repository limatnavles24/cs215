r(z,w) = \frac{\sum_{i=1}^{N}{(z_i-\mu_z)(w_i-\mu_w)}}{\sigma_z \cdot \sigma_w}\\
\\
\text{now if mean of } x_i = \mu_x \text{ then mean of } z_i(= ax_i + b) = a\mu_x + b \text{ similarly for } w_i.\\
\\
\text{similarly }\sigma_z = \left|a\right| \cdot \sigma_x \text{ and }\sigma_w = \left|c\right| \cdot \sigma_y \\
\\
r(z,w)=\frac{\sum_{i=1}^{N}{[(ax_i+b)-(a\mu_x+b)] \cdot [(cy_i+d)-(c\mu_y+d)]}}{\left|a\right|\sigma_x \cdot \left|c\right|\sigma_y} = \frac{\sum_{i=1}^{N}{a(x_i-\mu_x) \cdot c(y_i-\mu_y)}}{\left|a\right|\sigma_x \cdot \left|c\right|\sigma_y} = sgn(ac)\cdot\frac{\sum_{i=1}^{N}{(x_i-\mu_x)(y_i-\mu_y)}}{\sigma_x\sigma_y}
now r(z,w) = r(x,y) when a and c have the same sign, r(z,w) = -r(x,y) if they have opposite signs. \hfill \text{ Qed.}
